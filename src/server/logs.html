	coalesce( json_agg(t), '[]') AS results
	FROM (
	SELECT CONCAT('http://localhost:8029/minotaur/v0.9/Lines(', "line"."id", ')') AS "@iot.selfLink",

		"result"->'line' AS "result",

 FROM "line"
 WHERE "file_id"  IN ((SELECT "file"."id" FROM "file" WHERE file."name" = 't_aggregation_data_fungi_metadata.csv'))
ORDER BY "id"
 LIMIT 200
) as t
